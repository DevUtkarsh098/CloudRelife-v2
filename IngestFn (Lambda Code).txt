import json
import boto3
from datetime import datetime
import os

dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('IncidentDataTable')
sfn_client = boto3.client('stepfunctions')

def lambda_handler(event, context):
    try:
        print("Incoming event:", json.dumps(event))

        # Support both API Gateway and direct invocation
        if 'body' in event:
            body = json.loads(event['body'])
        else:
            body = event  # direct invoke or test payload

        # Validate required fields
        required_fields = ['incident_id', 'description', 'state', 'district', 'severity']
        for field in required_fields:
            if field not in body:
                return {
                    'statusCode': 400,
                    'body': json.dumps({'error': f'Missing required field: {field}'})
                }

        # Default timestamp if not provided
        timestamp = body.get('timestamp', datetime.utcnow().isoformat())

        # Build DynamoDB item
        item = {
            'incident_id': body['incident_id'],
            'description': body['description'],
            'state': body['state'],
            'district': body['district'],
            'severity': int(body['severity']),
            'timestamp': timestamp,
            'status': 'INGESTED'
        }

        # Save to DynamoDB
        table.put_item(Item=item)

        # Trigger Step Function
        state_machine_arn = os.environ['STATE_MACHINE_ARN']

        sfn_input = {
            "incident_id": body['incident_id'],
            "state": body['state'],
            "district": body['district'],
            "severity": body['severity'],
            "timestamp": timestamp,
            "description": body['description']
        }

        sfn_response = sfn_client.start_execution(
            stateMachineArn=state_machine_arn,
            input=json.dumps(sfn_input)
        )

        return {
            'statusCode': 200,
            'body': json.dumps({
                'message': 'Incident ingested and pipeline triggered successfully',
                'item': item,
                'step_function_execution_arn': sfn_response['executionArn']
            })
        }

    except Exception as e:
        print("Error:", str(e))
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }
